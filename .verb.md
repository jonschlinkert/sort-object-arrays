## Usage

```js
var sortArrays = require('{%= name %}');

var obj = sortArrays({foo: ['z', 'a', 'b']});
//=> obj.foo[0] === 'a'
//=> obj.foo[1] === 'b'
//=> obj.foo[2] === 'z'
```

Also sorts array elements on nested properties:

```js
var obj = sortArrays({foo: {bar: {baz: ['z', 'a', 'b']}}});
//=> obj.foo.bar.baz[0] === 'a'
//=> obj.foo.bar.baz[1] === 'b'
//=> obj.foo.bar.baz[2] === 'z'
```
